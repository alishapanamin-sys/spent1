<script>
(() => {
  // ‚úÖ Put your Monetizer Smartlink here:
  const SMARTLINK = "https://iconics.adsleads1.store/?utm_medium=24c9e8ead1a61abc4ca11515604da84dbebe69ba&utm_campaign=spent";

  function getParam(name){
    const u = new URL(window.location.href);
    return u.searchParams.get(name);
  }

  // ‚úÖ Updated segments (Your request):
  // - 1x Good Luck
  // - 1x Congrats iPhone 17
  // - 1x Congrats Top iPad
  // - 1x Congrats Samsung Ultra
  // - 1x Congrats Gift Card
  // - 2x No Prize
  // (Compliant phrasing: "chance to win")
  const segments = [
    { label: "GOOD LUCK NEXT TIME", kind:"luck", weight: 1 },

    { label: "CONGRATS! YOU UNLOCKED A CHANCE TO WIN IPHONE 17", kind:"win",  weight: 1 },
    { label: "CONGRATS! YOU UNLOCKED A CHANCE TO WIN TOP-TIER IPAD", kind:"win",  weight: 1 },
    { label: "CONGRATS! YOU UNLOCKED A CHANCE TO WIN SAMSUNG ULTRA (TOP MODEL)", kind:"win",  weight: 1 },
    { label: "CONGRATS! YOU UNLOCKED A GIFT CARD REWARD", kind:"win",  weight: 1 },

    { label: "NO PRIZE THIS TIME", kind:"lose", weight: 1 },
    { label: "NO PRIZE THIS TIME", kind:"lose", weight: 1 }
  ];

  // Drawing
  const canvas = document.getElementById("wheel");
  const ctx = canvas.getContext("2d");
  const spinBtn = document.getElementById("spinBtn");
  const resetBtn = document.getElementById("resetBtn");
  const resultBox = document.getElementById("result");
  const resultTitle = document.getElementById("resultTitle");
  const resultMsg = document.getElementById("resultMsg");
  const ctaRow = document.getElementById("ctaRow");
  const continueBtn = document.getElementById("continueBtn");
  const spinAgainBtn = document.getElementById("spinAgainBtn");

  const W = canvas.width, H = canvas.height;
  const cx = W/2, cy = H/2;
  const r = Math.min(W,H)*0.45;

  const colors = [
    "#f59e0b", "#22c55e", "#60a5fa", "#ef4444", "#a78bfa", "#f97316",
    "#34d399", "#fb7185"
  ];

  function drawWheel(angle=0){
    ctx.clearRect(0,0,W,H);

    ctx.save();
    ctx.translate(cx,cy);
    ctx.rotate(angle);

    const total = segments.length;
    const arc = (Math.PI*2)/total;

    for(let i=0;i<total;i++){
      const start = i*arc;
      const end = start + arc;

      ctx.beginPath();
      ctx.moveTo(0,0);
      ctx.arc(0,0,r,start,end);
      ctx.closePath();
      ctx.fillStyle = colors[i % colors.length];
      ctx.fill();

      ctx.strokeStyle = "rgba(255,255,255,.35)";
      ctx.lineWidth = 3;
      ctx.stroke();

      ctx.save();
      ctx.rotate(start + arc/2);
      ctx.textAlign = "right";
      ctx.fillStyle = "rgba(10, 15, 25, .92)";
      ctx.font = "800 26px system-ui, -apple-system, Segoe UI, Roboto, Arial";
      wrapText(segments[i].label, r - 22, 0, 340, 30);
      ctx.restore();
    }

    ctx.beginPath();
    ctx.arc(0,0,120,0,Math.PI*2);
    ctx.fillStyle = "rgba(0,0,0,.25)";
    ctx.fill();
    ctx.strokeStyle = "rgba(255,255,255,.2)";
    ctx.lineWidth = 6;
    ctx.stroke();

    ctx.restore();
  }

  function wrapText(text, x, y, maxWidth, lineHeight){
    const words = text.split(" ");
    let line = "";
    let yy = y;
    for(let n=0; n<words.length; n++){
      const testLine = line + words[n] + " ";
      const metrics = ctx.measureText(testLine);
      if(metrics.width > maxWidth && n > 0){
        ctx.fillText(line.trim(), x, yy);
        line = words[n] + " ";
        yy += lineHeight;
      } else {
        line = testLine;
      }
    }
    ctx.fillText(line.trim(), x, yy);
  }

  function weightedPick(items){
    const sum = items.reduce((a,b)=>a+b.weight,0);
    let rnd = Math.random() * sum;
    for(const it of items){
      rnd -= it.weight;
      if(rnd <= 0) return it;
    }
    return items[items.length-1];
  }

  let spinning = false;
  let currentAngle = 0;

  function buildSmartlink(){
    const u = new URL(SMARTLINK, window.location.origin);
    const passthrough = ["subid","sub_id","clickid","cid","utm_source","utm_campaign","utm_medium","utm_content"];
    const here = new URL(window.location.href);
    passthrough.forEach(k=>{
      const v = here.searchParams.get(k);
      if(v && !u.searchParams.has(k)) u.searchParams.set(k, v);
    });
    return u.toString();
  }

  function showResult(seg){
    resultBox.classList.add("show");
    ctaRow.style.display = "flex";
    continueBtn.href = buildSmartlink();

    resultTitle.textContent =
      seg.kind === "win"  ? "üéâ Congratulations!" :
      seg.kind === "luck" ? "üçÄ Good Luck!" :
                            "üòî Try Again";

    resultMsg.textContent = seg.label + ". Tap Continue to proceed.";
  }

  function spin(){
    if(spinning) return;

    if(!SMARTLINK || SMARTLINK === "YOUR_SMARTLINK_HERE"){
      alert('Please set your Monetizer Smartlink: SMARTLINK = "..."');
      return;
    }

    spinning = true;
    spinBtn.disabled = true;
    resetBtn.disabled = true;
    resultBox.classList.remove("show");
    ctaRow.style.display = "none";

    const chosen = weightedPick(segments);
    const idx = segments.indexOf(chosen);

    const total = segments.length;
    const arc = (Math.PI*2)/total;

    // Pointer at top. Land chosen slice center at -90 degrees.
    const sliceCenter = idx*arc + arc/2;
    const target = (Math.PI*1.5) - sliceCenter;

    const extraSpins = 6 + Math.random()*2; // 6-8 spins
    const startAngle = currentAngle;
    const endAngle = target + extraSpins*(Math.PI*2);

    const duration = 4200 + Math.random()*800;
    const startTime = performance.now();
    const easeOutCubic = t => 1 - Math.pow(1 - t, 3);

    function frame(now){
      const t = Math.min(1, (now - startTime)/duration);
      const eased = easeOutCubic(t);
      currentAngle = startAngle + (endAngle - startAngle)*eased;
      drawWheel(currentAngle);

      if(t < 1){
        requestAnimationFrame(frame);
      } else {
        spinning = false;
        resetBtn.disabled = false;
        showResult(chosen);
      }
    }
    requestAnimationFrame(frame);
  }

  function reset(){
    spinning = false;
    spinBtn.disabled = false;
    resetBtn.disabled = true;
    resultBox.classList.remove("show");
    ctaRow.style.display = "none";
  }

  spinBtn.addEventListener("click", spin);
  resetBtn.addEventListener("click", reset);
  spinAgainBtn.addEventListener("click", () => {
    reset();
    spin();
  });

  drawWheel(currentAngle);
})();
